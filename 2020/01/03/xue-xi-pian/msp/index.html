环境搭建

软件 版本 备注
centos 7.5
JDK 1.8
mysql 5.7+
redis 3.2+

JDK

yum install java-1.8.0-openjdk\* -y

mysql

设置mysql大小写不敏感
my.conf 加上 lower_case_table_names =1

## 下载yum源
wget wget https://dev.mysql.com/get/mysql80-community-release-el7-3.noarch.rpm
## 安装yum源
yum -y install mysql80-community-release-el7-3.noarch.rpm
## 查看源中激活的mysql版本
yum repolist all | grep mysql
## 修改源中需要安装的版本
yum-config-manager --disable mysql80-community
yum-config-manager --enable mysql57-community
或
手动编辑/etc/yum.repos.d/mysql-community.repo 文件来选择，把默认版本的enabled=1改为0，需要版本的enabled=1
## 安装mysql
yum -y install mysql-community-server
## 卸载yum源防止自动更新
yum -y remove mysql80-community-release-el7-3.noarch.rpm
## 查看系统服务
systemctl list-unit-files
## 设置开机启动
systemctl enable mysqld.service
## 启动MySQL服务
systemctl start mysqld.service
## 查看mysql默认密码：rcNZ-W4-2wiS
grep 'temporary password' /var/log/mysqld.log
## 登录并修改默认密码
mysql –u root –p
ALTER USER 'root'@'localhost' IDENTIFIED BY '!@#123QWe';
## 配置中增加大小写不敏感：lower_case_table_names =1
vim /etc/my.cnf
## 创建所有IP能访问的用户
create user 'mysql'@'%' identified by '123qweQ!';
## 开放远程登录权限
GRANT ALL PRIVILEGES ON *.* TO 'mysql'@'%' IDENTIFIED BY '123qweQ!' WITH GRANT OPTION;
FLUSH PRIVILEGES;

redis

## 下载redis
wget http://download.redis.io/releases/redis-4.0.2.tar.gz
## 解压并安装
tar xzf redis-4.0.2.tar.gz -C /usr/local/
## 进入redis根目录
cd redis-4.0.2
## 编译redis
make
## 安装redis到系统
make install
## 将初始化脚本放到/etc/init.d目录
cp redis源文件夹目录/utils/redis_init_script /etc/init.d/redis_端口号
## 创建目录
mkdir /etc/redis ## 存放Redis的配置文件
mkdir -p /var/redis/端口号 ## 存放Redis的持久化文件
## 修改配置文件
cp redis源文件夹目录/redis.conf /etc/redis/6379.conf
vim 6379.conf
# daemonize yes                         使Redis以守护进程模式运
# pidfile   /var/run/redis_端口号.pid    设置Redis的PID文件位置
# port      端口号                       设置Redis监听的端口号
# dir       /var/redis/端口号            设置持久化文件存放位置
## 启动关闭服务
/etc/init.d/redis_6379 start
/etc/init.d/redis_6379 stop
## 也可以通过systemctl来配置服务为系统服务，待完善
